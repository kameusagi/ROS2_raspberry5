【起動手順】
# ROS 2 Jazzy システム環境の読み込み
source /opt/ros/jazzy/setup.bash

# ビルド削除
rm -rf build/ install/ log/

# ビルド
colcon build --symlink-install #全部
colcon build --packages-select my_bringup #部分

# 自身のビルド結果を現在のターミナルに反映
source install/setup.bash

#　実行
・ros2 launch my_bringup led_system_launch.py #　複数ノードを一度に起動
・ros2 run dummy_sensor01 dummy_sensor_node　# パッケージ名と、setup.pyで定義したexecutable名を指定

【構成】
ros2_ws/
 ├── src/
 │    ├── sensor_interfaces/     # カスタムmsg定義
 │    ├── sensor_drivers/        # 各センサドライバ
 │    ├── sensor_core/           # センサ統合管理
 │    └── sensor_monitor/        # PC側受信ノード
 │    └── my_bringup/            # 複数ノードを一括起動するlaunch管理

【ROS2コマンド集】
　参考：https://qiita.com/NeK/items/33d62c61fc1a8c0c233e
    colcon build --symlink-install
    colcon build --symlink-install --packages-up-to [package1 package2 ...]
    colcon build --cmake-clean-cache
    ros2 pkg create my_node --build-type ament_python --dependencies rclpy

【スレッドのガントチャート】
# LTTng本体と関連ツールのインストール
sudo apt install lttng-tools lttng-modules-dkms liblttng-ust-dev python3-lttng

# ROS 2 追跡・解析用パッケージのインストール
sudo apt install ros-$ROS_DISTRO-tracetools-analysis ros-$ROS_DISTRO-ros2trace

rm -rf ~/.ros/tracing/my_tracing_session
ros2 trace start my_tracing_session
source install/setup.bash
ros2 launch my_bringup led_system_launch.py
1分ほど経ったら Ctrl + C で停止
ros2 trace stop my_tracing_session
# 生データを解析用データに変換
cd ~/.ros/tracing/my_tracing_session/ust #
ros2 run tracetools_analysis convert .
cd ~/Desktop/ROS2/ros2_ws
python3 analysis.py
